Почему код даёт именно такие результаты?
var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2
++ префиксная форма унарного оператора инкремента, увеличивает а на 1 и возвощает значение а+1 (2)

d = b++; alert(d); // 1
++ постфиксаня форма унарного оператора инкремента, увеличивает b на 1 и возвощает значение b до изменения (1)

c = (2+ ++a); alert(c); // 5
++a вернет 2+1 = 3, 2+3 =5

d = (2+ b++); alert(d); // 4
b++ вернет b, равное 2, т.о. 2+2=4

alert(a); // 3
alert(b); // 3
для a и b изначально равных 1 дважды был примене оператор инкремента
